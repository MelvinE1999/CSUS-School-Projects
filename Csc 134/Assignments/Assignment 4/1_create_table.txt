CREATE TABLE PROJECT
(
PROJECT_ID INTEGER,
TITLE VARCHAR(50),
ARCHIVED BOOLEAN,
PROJECT_STATUS CHAR(10),
PRIMARY KEY (PROJECT_ID)
);

CREATE TABLE LANGUAGE_USED
(
LANGUAGE VARCHAR(3),
PROJECT_ID INTEGER, 
PRIMARY KEY(PROJECT_ID,LANGUAGE),
FOREIGN KEY(PROJECT_ID) 
REFERENCES PROJECT (PROJECT_ID)
);

CREATE TABLE INVESTIGATOR
(
SSN INTEGER(9),
FIRST_NAME TEXT,
LAST_NAME TEXT,
ORGANIZATION TEXT,
ADDRESS TEXT,
PHONE CHAR(12),
PRIMARY KEY (SSN)
);

CREATE TABLE WORKS_ON
(
PROJECT_ID INTEGER ,
SSN INTEGER(9),
INV_ROLE TINYTEXT,
FOREIGN KEY (SSN) 
REFERENCES INVESTIGATOR (SSN),
FOREIGN KEY (PROJECT_ID) 
REFERENCES PROJECT(PROJECT_ID),
PRIMARY KEY(PROJECT_ID,SSN)
);

CREATE TABLE REVIEWER
(
SSN INTEGER(9),
FIRST_NAME TEXT,
LAST_NAME TEXT,
PRIMARY KEY (SSN)
);

CREATE TABLE REVIEWS
(
PROJECT_ID INTEGER,
SSN INTEGER(9),
REVIEWER_ROLE TINYTEXT,
FOREIGN KEY (PROJECT_ID) 
REFERENCES PROJECT (PROJECT_ID),
FOREIGN KEY (SSN) 
REFERENCES REVIEWER (SSN),
PRIMARY KEY(PROJECT_ID,SSN)
);

CREATE TABLE CONTACT_HISTORY
(
CONTACT_ID INTEGER,
PROJECT_ID INTEGER,
CP_NAME TEXT,
CONTACT_TYPE TINYTEXT,
CONTACT_DETAILS TEXT,
FOREIGN KEY (PROJECT_ID) 
REFERENCES PROJECT (PROJECT_ID),
PRIMARY KEY(CONTACT_ID,PROJECT_ID)
);
